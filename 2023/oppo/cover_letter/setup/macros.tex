%%%%%%%%%%%%%%% Title of the letter %%%%%%%%%%%%%%% 
\newcommand{\name}[2]{
    % Name
    \Huge
    \raggedright \textbf{#1} \par

    \vspace*{0.3em}
    
    % Profession
    \LARGE
    \raggedright #2 
    \normalsize \par \vspace{3em}
}

%%%%%%%%%%%%%%% Person that receives the ;etter %%%%%%%%%%%%%%% 
\newcommand{\receiver}[1]{
    \large
    \raggedright \textbf{To: #1}
    \normalsize \par \vspace{1em}
}

%%%%%%%%%%%%%%% Paragraph of a letter %%%%%%%%%%%%%%% 
\newcommand{\para}[1]{
    \large #1
    \normalsize \par \vspace{1em}
}

%%%%%%%%%%%%%%% Bottom of the letter %%%%%%%%%%%%%%% 
\newcommand{\bottom}[3]{
    %%% Contact info in the footer %%%
    \fancyfoot{
        \normalsize
        \raggedright #1 $\cdot$ #2 $\cdot$ #3

        %%% Draw a rule under the footer %%%
        \begin{tikzpicture}
            \fill[bar] (0, 0) rectangle (20,0.15);
        \end{tikzpicture}
    }
}
